(function () {
	'use strict';
	function getPushData(SEARCH_USER_ID) {
		let _pushData = [];
		let _userId = "";
		let _pushContent = "";
		let _searchIndex = -1;
		let _html = "";
		$(".push").each(function () {
			_userId = jQuery.trim($(this).find(".push-userid").text());
			_searchIndex = _pushData.findIndex((obj => obj.userId === _userId));
			if (_searchIndex !== -1) {
				_pushContent = _pushData[_searchIndex]["pushContent"];
				_pushContent += ($(this).find(".push-content").text()).replace(": ", "") + "</br>";
				_pushData[_searchIndex]["pushContent"] = _pushContent;
			} else {
				_pushData.push({
					userId: _userId,
					pushContent: ($(this).find(".push-content").text()).replace(": ", "") + "</br>"
				});
			}
		});
		if (jQuery.trim(SEARCH_USER_ID) !== "") {
			_searchIndex = _pushData.findIndex((obj => obj.userId === SEARCH_USER_ID));
			_html += "【" + _pushData[_searchIndex]["userId"] + "】</br>" + _pushData[_searchIndex]["pushContent"];
		} else {
			let i = 0;
			let iMax = _pushData.length;
			let _text = "";
			for (i; i < iMax; i++) {
				_text += "# " + (i + 1) + "【" + _pushData[i]["userId"] + "】</br>" + _pushData[i]["pushContent"] + "</br>";
			}
			_html = "<!DOCTYPE html><body>" + _text + "</body>";
		}
		if (_html !== "") {
			window.open('blankPage.html').document.write(_html);
		}
	}
	getPushData();
})();
